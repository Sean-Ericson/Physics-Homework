    \begin{align*}

    \end{align*}

        \Tr[\rho\sigma_x] &= \frac{1}{2}\Tr[\mqty(1+c & a-ib \\ a+ib & 1-c)\mqty(0&1\\1&0)] = \frac{1}{2}(a - ib + a + ib) = a \\

    \end{align*}

    \item In the ``I Know Nothing'' state,
    \[ \ev{\sigma_x} = \ev{\sigma_y} = \ev{\sigma_z} = 0 \]
