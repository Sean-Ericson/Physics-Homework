    & = \frac{1}{3}M_{\mu\mu}\delta_{\alpha\beta} + \frac{1}{4}\epsilon_{\mu\sigma\tau}(M_{\sigma\tau} - M_{\tau\sigma})\epsilon_{\mu\alpha\beta} + \frac{1}{2}(M_{\alpha\beta} + M_{\beta\alpha}) - \frac{1}{3}M_{\mu\mu}\delta_{\alpha\beta} \\
